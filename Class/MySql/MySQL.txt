-Database Operations:
create table, insert table, select, update,delete

//Create table
CREATE TABLE student(id NUMBER, name VARCHAR, email VARCHAR, course VARCHAR);

//INSERT TABLE
INSERT INTO student(id, name, email, course) VALUES (306, "Ankita", "ankita@gmail.com", "AIML");

//select table
SELECT * FROM student

//update table
UPDATE student SET name = "ANkita Mane" WHERE id = 306;

//Delete table
DELETE FROM student WHERE id = 306;
